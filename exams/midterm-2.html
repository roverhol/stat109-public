### Scenario C

**Study.** In Feburary of 2024, adults with mild hypertension who visited a particular rural clinic were randomly assigned to a daily walking coach app (Treatment) or a pamphlet (Control). After 8 weeks, systolic blood pressure (mmHg) was measured once per person.

### Scenario D

**Study.** Data was collected from first-year students at Cal Poly Humboldt in a section of calculus who chose to enroll in an optional calculus study workshop (Yes/No) in Fall of 2025. The final exam score in was recorded for all first-year students in that section of Calculus at the end of term.

**Research question.** What is the difference in mean final exam score for all first year students in Calculus who chose to enroll in the workshop and don't?

```{r}
#| label: data-D
set.seed(2404)
n <- 90
fulljob <- rbinom(n, 1, 0.45)
workshop <- rbinom(n, 1, 0.5 - 0.25 * fulljob)
score <- round(78 - 10 * fulljob + 5 * workshop + rnorm(n, 0, 5), 1)
D <- tibble(
  workshop = factor(ifelse(workshop == 1, "Yes", "No"), levels = c("No", "Yes")),
  exam_score = score
)
D |> head(12)
D |> 
    group_by(workshop) |> 
    summarise(
        n = n(), 
        mean = mean(exam_score), 
        sd = sd(exam_score)
    )
D |> ggplot(aes(x=exam_score, fill=workshop)) +
     geom_histogram(bins=8, alpha=0.5) +
     ylab("Number of Students") +
     xlab("Final Exam Score")
t.test(exam_score ~ workshop, data = D, var.equal = FALSE)
D$fulljob = as.factor(fulljob)
levels(D$fulljob) = c("No fulltime job", "Fulltime job")
D |> 
    group_by(fulljob, workshop) |> 
    summarise(
        n = n(), 
        mean = mean(exam_score), 
        sd = sd(exam_score)
    )
D |> ggplot(aes(x=exam_score, fill=workshop)) +
     geom_histogram(bins=8, alpha=0.5) +
     ylab("Number of Students") +
     xlab("Final Exam Score") + 
     facet_wrap(~fulljob)
```

### Scenario G

**Study.** In a study of flu vaccines, 120 healthy adult volunteers were randomly assigned to vaccine or placebo in the fall of 2024. Whether or not they got the flu (yes/no) during the next 3 months of follow-up was recorded.

**Research question.** Does the population rate of flu differ between adults who get the vaccine or not?

```{r}
#| label: data-G
set.seed(2407)
n <- 120
G <- tibble(
  arm = factor(rep(c("Placebo", "Vaccine"), each = n), levels = c("Placebo", "Vaccine")),
  infected = c(rbinom(n, 1, 0.22), rbinom(n, 1, 0.09))
)



xa <- sum(G$infected[G$arm == "Placebo"]); na <- sum(G$arm == "Placebo")
xb <- sum(G$infected[G$arm == "Vaccine"]); nb <- sum(G$arm == "Vaccine")
prop.test(x = c(xa, xb), n = c(na, nb), correct = FALSE)
G$flu = as.factor(G$infected)
levels(G$flu) = c("No", "Yes")
G |> 
   group_by(arm) |>
   count(flu)

G |>
  count(arm, flu) |>
  ggplot(aes(x = arm, y = n, fill = flu)) +
  geom_col(position = "fill") +
  scale_y_continuous(labels = scales::percent_format()) +
  labs(
    title = "Flu outcome proportions by study arm",
    x = "Study arm",
    y = "Proportion within arm",
```

### Scenario J

**Study.** A healthcare system manager obtained records from four clinics (A, B, C, D) on whether their patients returned for a follow-up visit within 30 days (yes/no) during 2025.

**Research question.** Is return rate differ between at least two of the four clinics for all patients in a similar time frame?

```{r}
#| label: data-J
set.seed(2410)
J <- tibble(
  clinic = factor(
    sample(c("A", "B", "C", "D"), 220, replace = TRUE),
    levels = c("A", "B", "C", "D")
  )
) |>
  mutate(
    p_ret = case_when(
      clinic == "A" ~ 0.72,
      clinic == "B" ~ 0.65,
      clinic == "C" ~ 0.58,
      TRUE ~ 0.70
    ),
    returned = rbinom(n(), 1, prob = p_ret)
  ) |>
  select(-p_ret)

J$returned = as.factor(J$returned)
levels(J$returned) = c("No", "Yes")

J |> 
   group_by(clinic) |>
   count(returned)

J |>
  count(clinic, returned) |>
  ggplot(aes(x = clinic, y = n, fill = returned)) +
  geom_col(position = "fill") +
  scale_y_continuous(labels = scales::percent_format()) +
  labs(
    title = "Return rates by clinic",
    x = "Clinic",
    y = "Proportion who returned within clinic",
    fill = "Return rate"
  )
tabJ = table(J$clinic, J$returned)
chisq.test(tabJ)
```
